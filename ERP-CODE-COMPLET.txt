function run(sql, params = []) {
  return new Promise((resolve, reject) => {
    db.run(sql, params, function (err) {
      if (err) {
        reject(err);
      } else {
        resolve(this);
      }
    });
  });
}

function get(sql, params = []) {
  return new Promise((resolve, reject) => {
    db.get(sql, params, (err, row) => {
      if (err) reject(err);
      else resolve(row);
    });
  });
}

function all(sql, params = []) {
  return new Promise((resolve, reject) => {
    db.all(sql, params, (err, rows) => {
      if (err) reject(err);
      else resolve(rows);
    });
  });
}

function resolveSafariLogoPath() {
  const candidates = [
    path.join(__dirname, 'public', 'logo_safari_constructions.png'),
    path.join(__dirname, 'public', 'logo_safari_constructions.jpg'),
    path.join(__dirname, 'public', 'logo_safari_constructions.jpeg'),
    path.join(__dirname, 'public', 'logo_safari_constructions.webp'),
    path.join(__dirname, 'logo_safari_constructions.png'),
    path.join(__dirname, 'logo_safari_constructions.jpg'),
    path.join(__dirname, 'logo_safari_constructions.jpeg'),
    path.join(__dirname, 'logo_safari_constructions.webp'),
  ];

  for (const filePath of candidates) {
    if (fs.existsSync(filePath)) {
      return filePath;
    }
  }

  const publicDir = path.join(__dirname, 'public');
  if (fs.existsSync(publicDir)) {
    const match = fs.readdirSync(publicDir).find(name =>
      /safari.*construct|construct.*safari/i.test(name) && /\.(png|jpg|jpeg|webp)$/i.test(name)
    );
    if (match) {
      return path.join(publicDir, match);
    }
  }

  return null;
}

async function enrichPurchaseOrders(orders) {
  if (!orders.length) {
    return [];
  }

  const ids = orders.map(order => Number(order.id)).filter(id => Number.isInteger(id));
  if (!ids.length) {
    return orders;
  }

  const placeholders = ids.map(() => '?').join(',');
  const itemRows = await all(`
    SELECT poi.*, mr.itemName as requestItemName, p.nomProjet, p.numeroMaison
    FROM purchase_order_items poi
    LEFT JOIN material_requests mr ON mr.id = poi.materialRequestId
    LEFT JOIN projects p ON p.id = mr.projetId
    WHERE poi.purchaseOrderId IN (${placeholders})
    ORDER BY poi.id ASC
  `, ids);

  const itemsByOrderId = {};
  itemRows.forEach(row => {
    const orderId = Number(row.purchaseOrderId);
    if (!itemsByOrderId[orderId]) {
      itemsByOrderId[orderId] = [];
    }
    itemsByOrderId[orderId].push({
      id: row.id,
      materialRequestId: row.materialRequestId,
      article: row.article,
      details: row.details,
      quantite: Number(row.quantite || 0),
      prixUnitaire: Number(row.prixUnitaire || 0),
      totalLigne: Number(row.totalLigne || 0),
      projetNom: row.nomProjet || null,
      numeroMaison: row.numeroMaison || null,
      requestItemName: row.requestItemName || null,
    });
  });

  return orders.map(order => {
    const items = itemsByOrderId[Number(order.id)] || [];
    const firstItem = items[0] || null;
    const totalFromItems = items.reduce((sum, item) => sum + Number(item.totalLigne || 0), 0);
    const projectNames = Array.from(new Set(items.map(item => item.projetNom).filter(Boolean)));
    const houseNumbers = Array.from(new Set(items.map(item => String(item.numeroMaison || '').trim()).filter(Boolean)));

    return {
      ...order,
      items,
      itemName: firstItem ? firstItem.article : (order.itemName || 'Article'),
      quantiteCommandee: firstItem ? Number(firstItem.quantite || 0) : Number(order.quantiteCommandee || 0),
      prixUnitaire: firstItem ? Number(firstItem.prixUnitaire || 0) : Number(order.prixUnitaire || 0),
      montantTotal: Number(order.montantTotal || totalFromItems || 0),
      nomProjet: projectNames.length ? projectNames.join(', ') : (order.nomProjet || '-'),
      numeroMaison: houseNumbers.length ? houseNumbers.join(', ') : '-',
      projetId: order.projetId || null,
    };
  });
}

  const [enriched] = await enrichPurchaseOrders([row]);

function renderPurchaseOrderPdf(doc, order) {
  const total = Number(order.montantTotal || 0);

  doc.font('Helvetica');
  const logoPath = resolveSafariLogoPath();
  if (logoPath) {
    doc.image(logoPath, 40, 30, { fit: [200, 76], align: 'left', valign: 'top' });
  } else {
    doc.rect(40, 30, 200, 76).stroke();
    doc.fontSize(11).text('LOGO SAFARI', 95, 62, { width: 90, align: 'center' });
  }

  doc.font('Helvetica-Bold').fontSize(24).fillColor('#000000').text('SAFARI CONSTRUCTIONS', 250, 56);
  doc.font('Helvetica-Bold').fontSize(30).fillColor('#006633').text('BON DE COMMANDE', 40, 134);
  doc.fillColor('#000000');

  const infoY = 186;
  doc.font('Helvetica').fontSize(12);
  doc.text(`Nom : ${String(order.fournisseur || '').trim() || '________________________________'}`, 40, infoY);
  doc.text('Telephone : __________________________', 40, infoY + 24);
  doc.text('Email : ______________________________', 40, infoY + 48);
  doc.text('Adresse : ____________________________', 40, infoY + 72);

  const tableTop = infoY + 120;
  const startX = 40;
  const widths = [145, 145, 145, 80];
  const headers = ['Article', 'Prix', 'Quantite', 'Total'];
  const rowHeight = 24;

  function drawGridRow(y, values, isHeader = false) {
    let x = startX;
    values.forEach((value, index) => {
      doc.rect(x, y, widths[index], rowHeight).stroke();
      doc.font(isHeader ? 'Helvetica-Bold' : 'Helvetica').fontSize(11).text(String(value), x + 8, y + 6, {
        width: widths[index] - 16,
        ellipsis: true,
      });
      x += widths[index];
    });
  }

  drawGridRow(tableTop, headers, true);

  const itemRows = (order.items && order.items.length ? order.items : [
    {
      article: order.itemName || 'Article',
      quantite: Number(order.quantiteCommandee || 0),
      prixUnitaire: Number(order.prixUnitaire || 0),
      totalLigne: Number(order.montantTotal || 0),
    },
  ]).map(item => [
    item.article || 'Article',
    `${Number(item.prixUnitaire || 0).toFixed(2)}`,
    Number(item.quantite || 0).toFixed(2),
    `${Number(item.totalLigne || 0).toFixed(2)}`,
  ]);

  const minRows = 7;
  while (itemRows.length < minRows) {
    itemRows.push(['', '', '', '']);
  }

  itemRows.forEach((row, idx) => drawGridRow(tableTop + rowHeight * (idx + 1), row, false));

  doc.font('Helvetica-Bold').fontSize(12).text(`Total : ${total.toFixed(2)} EUR`, 40, tableTop + rowHeight * (itemRows.length + 1) + 20);
  doc.font('Helvetica-Oblique').fontSize(11).text('Safari Constructions - Professionnalisme & Qualite', 40, 770, {
    align: 'center',
    width: 515,
  });
}

async function generatePurchaseOrderPdfBuffer(order) {
  return await new Promise((resolve, reject) => {
    const chunks = [];
    const doc = new PDFDocument({ size: 'A4', margin: 40 });
    doc.on('data', chunk => chunks.push(chunk));
    doc.on('end', () => resolve(Buffer.concat(chunks)));
    doc.on('error', reject);
    renderPurchaseOrderPdf(doc, order);
    doc.end();
  });
}

async function archivePurchaseOrderPdf(purchaseOrderId) {
  const order = await getPurchaseOrderById(purchaseOrderId);
  if (!order) {
    return null;
  }

  const sectionCode = 'achats';
  const supplierSlug = String(order.fournisseur || 'fournisseur')
    .replace(/[^a-z0-9]+/gi, '-')
    .replace(/^-+|-+$/g, '')
    .toLowerCase() || 'fournisseur';
  const fileName = `bon-commande-${purchaseOrderId}-${supplierSlug}.pdf`;
  const relativePath = path.join(sectionCode, fileName);
  const directory = path.join(ARCHIVE_ROOT, sectionCode);
  const absolutePath = path.join(ARCHIVE_ROOT, relativePath);
  fs.mkdirSync(directory, { recursive: true });

  const buffer = await generatePurchaseOrderPdfBuffer(order);
  await fs.promises.writeFile(absolutePath, buffer);

  await run('DELETE FROM generated_documents WHERE entityType = ? AND entityId = ? AND sectionCode = ?', ['purchase_order', purchaseOrderId, sectionCode]);
  await run(
    'INSERT INTO generated_documents (sectionCode, sectionLabel, entityType, entityId, title, fileName, relativePath, createdAt, updatedAt) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)',
    [sectionCode, 'Achats', 'purchase_order', purchaseOrderId, `Bon de commande #${purchaseOrderId}`, fileName, relativePath, new Date().toISOString(), new Date().toISOString()]
  );

  return {
    title: `Bon de commande #${purchaseOrderId}`,
    fileName,
    fileUrl: `/archives/${relativePath.replace(/\\/g, '/')}`,
  };
}

function getArchiveSectionLabel(sectionCode) {

    achats: 'Achats',

function sanitizeFileName(fileName) {
  return String(fileName || 'document')
    .replace(/[<>:"/\\|?*\x00-\x1F]/g, '-')
    .replace(/\s+/g, '-')
    .replace(/-+/g, '-')
    .replace(/^-+|-+$/g, '')
    .slice(0, 120) || 'document';
}

  const safeNumber = Number(numberValue) || 0;

}

async function archiveUploadedDocument({ sectionCode, title, fileName, fileBuffer, projectId = null }) {
  const safeSection = ['achats', 'construction', 'comptabilite'].includes(sectionCode) ? sectionCode : 'construction';
  const safeName = sanitizeFileName(fileName || 'document');
  const stamp = new Date().toISOString().replace(/[:.]/g, '-');
  const finalFileName = `${stamp}-${safeName}`;
  const relativePath = getArchiveRelativePath(safeSection, finalFileName);
  const absoluteDir = path.join(ARCHIVE_ROOT, safeSection);
  const absolutePath = path.join(ARCHIVE_ROOT, relativePath);
  fs.mkdirSync(absoluteDir, { recursive: true });

  const now = new Date().toISOString();
  const numericProjectId = Number(projectId || 0);
  const linkedEntityId = Number.isFinite(numericProjectId) && numericProjectId > 0 ? numericProjectId : 0;

    'INSERT INTO generated_documents (sectionCode, sectionLabel, entityType, entityId, title, fileName, relativePath, createdAt, updatedAt) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)',
    [safeSection, getArchiveSectionLabel(safeSection), 'manual_upload', linkedEntityId, title || safeName, finalFileName, relativePath, now, now]
  );

  return {
    id: result.lastID,
    sectionCode: safeSection,
    title: title || safeName,
    fileName: finalFileName,
    fileUrl: `/archives/${relativePath.replace(/\\/g, '/')}`,
  };
}

async function preparePurchaseOrderItems(payload) {
  const {
    items,
    materialRequestId,
    quantiteCommandee,
    prixUnitaire,
    montantTotal,
  } = payload;

  let normalizedItems = [];
  if (Array.isArray(items) && items.length > 0) {
    normalizedItems = items;
  } else if (materialRequestId) {
    normalizedItems = [{
      materialRequestId,
      quantite: quantiteCommandee,
      prixUnitaire,
      article: '',
      details: '',
      totalLigne: montantTotal,
    }];
  }

  if (!normalizedItems.length) {
    return [];
  }

  const preparedItems = [];
  for (const rawItem of normalizedItems) {
    const requestId = Number(rawItem.materialRequestId || rawItem.requestId || materialRequestId);
    const request = await get('SELECT id, projetId, itemName, description FROM material_requests WHERE id = ?', [requestId]);
    if (!request) {
      return null;
    }

    const quantity = Number(rawItem.quantite ?? rawItem.quantity ?? quantiteCommandee);
    const unitPrice = Number(rawItem.prixUnitaire ?? rawItem.unitPrice ?? prixUnitaire);
    if (Number.isNaN(quantity) || quantity <= 0 || Number.isNaN(unitPrice) || unitPrice < 0) {
      return 'INVALID_PO_ITEM';
    }

    const article = String(rawItem.article || request.itemName || '').trim() || 'Article';
    const details = String(rawItem.details || request.description || '').trim();
    const lineTotal = Number(rawItem.totalLigne ?? quantity * unitPrice);

    preparedItems.push({
      materialRequestId: requestId,
      projetId: request.projetId,
      article,
      details,
      quantite: quantity,
      prixUnitaire: unitPrice,
      totalLigne: Number.isNaN(lineTotal) ? quantity * unitPrice : lineTotal,
    });
  }

  return preparedItems;
}

async function syncExpensesForPurchaseOrder(purchaseOrderId, override = {}) {
  try {
    await run('DELETE FROM expenses WHERE purchaseOrderId = ?', [purchaseOrderId]);
  } catch (e) {}

  const order = await getPurchaseOrderById(purchaseOrderId);
  if (!order || !order.items || !order.items.length) {
    return;
  }

  const expColRows = await all('PRAGMA table_info(expenses)');
  const expCols = new Set(expColRows.map(row => row.name));
  const dateStr = override.dateCommande || order.dateCommande || new Date().toISOString();
  const supplier = String(override.fournisseur || order.fournisseur || '').trim();
  const createdBy = override.createdBy || 'admin';

  for (const item of order.items) {
    let itemProjetId = override.projetId || item.projetId || order.projetId || null;
    if (!itemProjetId && item.materialRequestId) {
      const request = await get('SELECT projetId FROM material_requests WHERE id = ?', [item.materialRequestId]);
      if (request) itemProjetId = request.projetId;
    }

    const columns = [];
    const values = [];
    const pushColumn = (columnName, value) => {
      if (expCols.has(columnName)) {
        columns.push(columnName);
        values.push(value);
      }
    };

    pushColumn('materialId', item.materialRequestId || null);
    pushColumn('projetId', itemProjetId);
    pushColumn('projectId', itemProjetId);
    pushColumn('description', item.article);
    pushColumn('item', item.article);
    pushColumn('quantite', Number(item.quantite || 0));
    pushColumn('quantity', Number(item.quantite || 0));
    pushColumn('prixUnitaire', Number(item.prixUnitaire || 0));
    pushColumn('unitPrice', Number(item.prixUnitaire || 0));
    pushColumn('montantTotal', Number(item.totalLigne || 0));
    pushColumn('totalPrice', Number(item.totalLigne || 0));
    pushColumn('dateExpense', dateStr);
    pushColumn('date', dateStr);
    pushColumn('fournisseur', supplier);
    pushColumn('supplier', supplier);
    pushColumn('categorie', 'materiaux');
    pushColumn('category', 'materiaux');
    pushColumn('statut', 'EN_ATTENTE');
    pushColumn('status', 'EN_ATTENTE');
    pushColumn('purchaseOrderId', purchaseOrderId);
    pushColumn('createdBy', createdBy);

    if (!columns.length) {
      continue;
    }

    const placeholders = columns.map(() => '?').join(', ');
    await run(`INSERT INTO expenses (${columns.join(', ')}) VALUES (${placeholders})`, values);
  }
}

async function reconcilePurchaseOrderExpenses() {
  const orders = await all('SELECT id FROM purchase_orders ORDER BY id ASC');
  for (const order of orders) {
    const countRow = await get('SELECT COUNT(*) as count FROM expenses WHERE purchaseOrderId = ?', [order.id]);
    if (!countRow || !countRow.count) {
      await syncExpensesForPurchaseOrder(order.id);
    }
  }
}

function authenticateToken(req, res, next) {
  const authHeader = req.headers.authorization;
  if (!authHeader) {
    return res.status(401).json({ error: 'Token manquant' });
  }

  const token = authHeader.split(' ')[1];
  if (!token) {
    return res.status(401).json({ error: 'Token manquant' });
  }

  jwt.verify(token, JWT_SECRET, (err, user) => {
    if (err) {
      return res.status(401).json({ error: 'Token invalide' });
    }
    req.user = user;
    next();
  });
}

app.post('/api/auth/login', async (req, res) => {
  const { username, password } = req.body;
  if (!username || !password) {
    return res.status(400).json({ error: 'Nom d\'utilisateur et mot de passe sont obligatoires' });
  }

  const user = await get('SELECT * FROM users WHERE username = ?', [username]);
  if (!user) {
    return res.status(401).json({ error: 'Utilisateur ou mot de passe invalide' });
  }

  const valid = await bcrypt.compare(password, user.password);
  if (!valid) {
    return res.status(401).json({ error: 'Utilisateur ou mot de passe invalide' });
  }

  const token = jwt.sign({ id: user.id, username: user.username, role: user.role }, JWT_SECRET, {
    expiresIn: '6h'
  });

  res.json({ token, username: user.username });
});

app.post('/api/purchase-orders', async (req, res) => {
  const {
    materialRequestId,
    creePar = 'admin',
    fournisseur,
    quantiteCommandee,
    prixUnitaire,
    montantTotal,
    dateLivraisonPrevue,
    dateCommande,
    items,
  } = req.body;

  if (!fournisseur || !String(fournisseur).trim()) {
    return res.status(400).json({ error: 'Le fournisseur est obligatoire' });
  }

  const preparedItems = await preparePurchaseOrderItems({
    items,
    materialRequestId,
    quantiteCommandee,
    prixUnitaire,
    montantTotal,
  });

  if (preparedItems === 'INVALID_PO_ITEM') {
    return res.status(400).json({ error: 'Chaque article doit avoir une quantite et un prix valides' });
  }

  if (!preparedItems || !preparedItems.length) {
    if (preparedItems === null) {
      return res.status(404).json({ error: 'Demande de materiel introuvable pour un article' });
    }
    return res.status(400).json({ error: 'Ajoute au moins un article pour le bon de commande' });
  }

  const firstRequestId = preparedItems[0].materialRequestId;
  const computedTotal = preparedItems.reduce((sum, item) => sum + Number(item.totalLigne || 0), 0);

  const result = await run(
    'INSERT INTO purchase_orders (materialRequestId, creePar, fournisseur, montantTotal, quantiteCommandee, prixUnitaire, dateCommande, dateLivraisonPrevue, statut, statutValidation) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)',
    [
      firstRequestId,
      String(creePar || 'admin').trim() || 'admin',
      String(fournisseur).trim(),
      computedTotal,
      preparedItems[0].quantite,
      preparedItems[0].prixUnitaire,
      dateCommande ? new Date(dateCommande).toISOString() : new Date().toISOString(),
      dateLivraisonPrevue || null,
      'EN_COURS',
      'EN_COURS',
    ]
  );

  for (const item of preparedItems) {
    await run(
      'INSERT INTO purchase_order_items (purchaseOrderId, materialRequestId, article, details, quantite, prixUnitaire, totalLigne) VALUES (?, ?, ?, ?, ?, ?, ?)',
      [result.lastID, item.materialRequestId, item.article, item.details, item.quantite, item.prixUnitaire, item.totalLigne]
    );
  }

  await syncExpensesForPurchaseOrder(result.lastID, {
    fournisseur: String(fournisseur).trim(),
    dateCommande: dateCommande ? new Date(dateCommande).toISOString() : new Date().toISOString(),
    createdBy: req.user ? req.user.username : 'admin',
  });
  await archivePurchaseOrderPdf(result.lastID);

  const order = await getPurchaseOrderById(result.lastID);
  res.status(201).json(order);
});

      dateCommande ? new Date(dateCommande).toISOString() : null,
      id,
    ]
  );

  if (result.changes === 0) {
    return res.status(404).json({ error: 'Bon de commande non trouve' });
  }

  if (preparedItems && preparedItems.length) {
    await run('DELETE FROM purchase_order_items WHERE purchaseOrderId = ?', [id]);
    for (const item of preparedItems) {
      await run(
        'INSERT INTO purchase_order_items (purchaseOrderId, materialRequestId, article, details, quantite, prixUnitaire, totalLigne) VALUES (?, ?, ?, ?, ?, ?, ?)',
        [id, item.materialRequestId, item.article, item.details, item.quantite, item.prixUnitaire, item.totalLigne]
      );
    }
  }

  await syncExpensesForPurchaseOrder(id, {
    fournisseur: String(fournisseur).trim(),
    dateCommande: dateCommande ? new Date(dateCommande).toISOString() : undefined,
    createdBy: req.user ? req.user.username : 'admin',
  });
  await archivePurchaseOrderPdf(id);

  const order = await getPurchaseOrderById(id);
  res.json(order);
});

app.get('/api/database-documents', async (req, res) => {
  const sectionCode = String(req.query.section || '').trim();
  const rows = sectionCode
    ? await all('SELECT * FROM generated_documents WHERE sectionCode = ? ORDER BY updatedAt DESC, id DESC', [sectionCode])
    : await all('SELECT * FROM generated_documents ORDER BY updatedAt DESC, id DESC');

  const normalizedRows = await Promise.all(rows.map(async row => {
    let resolvedTitle = String(row.title || '').trim();
    const entityType = String(row.entityType || '').trim().toLowerCase();
    const entityId = Number(row.entityId || 0);
    let projectId = 0;

    if ((entityType === 'manual_upload' || entityType === 'upload') && Number.isFinite(entityId) && entityId > 0) {
      projectId = entityId;
    } else if (entityType === 'purchase_order' && Number.isFinite(entityId) && entityId > 0) {
      const order = await get('SELECT projetId FROM purchase_orders WHERE id = ?', [entityId]);
      projectId = Number(order?.projetId || 0);
    } else if (entityType === 'revenue' && Number.isFinite(entityId) && entityId > 0) {
      const revenue = await get('SELECT projetId FROM revenues WHERE id = ?', [entityId]);
      projectId = Number(revenue?.projetId || 0);
    }

    if (Number.isFinite(projectId) && projectId > 0) {
      const project = await get('SELECT nomProjet, numeroMaison FROM projects WHERE id = ?', [projectId]);
      if (project) {
        const projectName = String(project.nomProjet || 'Projet').trim() || 'Projet';
        const houseNumber = String(project.numeroMaison || '').trim() || '-';
        resolvedTitle = `${projectName}-${houseNumber}`;
      }
    }

    if (!resolvedTitle) {
      resolvedTitle = String(row.fileName || 'Document').trim() || 'Document';
    }

    resolvedTitle = resolvedTitle.replace(/\s*-\s*/g, '-').replace(/\s+/g, ' ').trim();

    if (resolvedTitle && resolvedTitle !== String(row.title || '').trim()) {
      await run('UPDATE generated_documents SET title = ?, updatedAt = ? WHERE id = ?', [resolvedTitle, new Date().toISOString(), row.id]);
    }

    return {
      ...row,
      title: resolvedTitle,
      fileUrl: `/archives/${String(row.relativePath || '').replace(/\\/g, '/')}`,
    };
  }));

  res.json(normalizedRows);
});

app.post('/api/expenses', async (req, res) => {
  const {
    materialId,
    projetId,
    description = '',
    quantite,
    prixUnitaire,
    fournisseur = '',
    categorie,
    item,
    category,
    quantity,
    unitPrice,
    projectId,
  } = req.body;

  const expenseDescription = description || item || '';
  const expenseCategory = categorie || category;
  const expenseQuantity = Number(quantite ?? quantity);
  const expenseUnitPrice = Number(prixUnitaire ?? unitPrice);
  const expenseProjectId = projetId || projectId || null;

  if (!expenseDescription || !expenseQuantity || !expenseUnitPrice || !expenseCategory) {
    return res.status(400).json({ error: 'Champs obligatoires manquants' });
  }

  const montantTotal = expenseQuantity * expenseUnitPrice;

  // Legacy databases can have old NOT NULL columns (category/item/quantity/unitPrice/totalPrice/projectId).
  // Build the INSERT dynamically to fill both old and new schemas when columns exist.
  const columnRows = await all('PRAGMA table_info(expenses)');
  const availableColumns = new Set(columnRows.map(row => row.name));
  const insertColumns = [];
  const insertValues = [];

  function pushColumnIfExists(columnName, value) {
    if (availableColumns.has(columnName)) {
      insertColumns.push(columnName);
      insertValues.push(value);
    }
  }

  pushColumnIfExists('materialId', materialId || null);
  pushColumnIfExists('projetId', expenseProjectId);
  pushColumnIfExists('projectId', expenseProjectId);
  pushColumnIfExists('description', expenseDescription);
  pushColumnIfExists('item', expenseDescription);
  pushColumnIfExists('quantite', expenseQuantity);
  pushColumnIfExists('quantity', expenseQuantity);
  pushColumnIfExists('prixUnitaire', expenseUnitPrice);
  pushColumnIfExists('unitPrice', expenseUnitPrice);
  pushColumnIfExists('montantTotal', montantTotal);
  pushColumnIfExists('totalPrice', montantTotal);
  pushColumnIfExists('dateExpense', new Date().toISOString());
  pushColumnIfExists('date', new Date().toISOString());
  pushColumnIfExists('fournisseur', fournisseur);
  pushColumnIfExists('supplier', fournisseur);
  pushColumnIfExists('categorie', expenseCategory);
  pushColumnIfExists('category', expenseCategory);
  pushColumnIfExists('statut', 'EN_ATTENTE');
  pushColumnIfExists('status', 'EN_ATTENTE');
  pushColumnIfExists('createdBy', req.user.username);

  const placeholders = insertColumns.map(() => '?').join(', ');
  const sql = `INSERT INTO expenses (${insertColumns.join(', ')}) VALUES (${placeholders})`;
  const result = await run(sql, insertValues);

  const expense = await get('SELECT * FROM expenses WHERE id = ?', [result.lastID]);
  res.status(201).json(expense);
});

    function escapeHtml(value) {
      return String(value ?? '')
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')

    function formatDate(value) {
      if (!value) return 'Non disponible';
      const date = new Date(value);
      if (Number.isNaN(date.getTime())) return 'Non disponible';
      return date.toLocaleDateString('fr-FR', { year: 'numeric', month: 'long', day: 'numeric' });
    }

    function formatDateWithSameDayTime(value) {
      if (!value) return 'Non disponible';
      const date = new Date(value);
      if (Number.isNaN(date.getTime())) return 'Non disponible';

      const now = new Date();
      const isSameDay = date.getFullYear() === now.getFullYear()
        && date.getMonth() === now.getMonth()
        && date.getDate() === now.getDate();

      if (isSameDay) {
        const dayPart = date.toLocaleDateString('fr-FR', { year: 'numeric', month: 'long', day: 'numeric' });
        const timePart = date.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' });
        return `${dayPart} à ${timePart}`;
      }

      return date.toLocaleDateString('fr-FR', { year: 'numeric', month: 'long', day: 'numeric' });
    }

    function formatDateForInput(value) {
      const date = value ? new Date(value) : new Date();
      if (Number.isNaN(date.getTime())) {
        return new Date().toISOString().slice(0, 10);

      const localDate = new Date(date.getTime() - date.getTimezoneOffset() * 60000);

    function computeProjectDisplayNameMap(projects) {
      const groups = new Map();
      projects.forEach(project => {
        const key = getProjectGroupKey(project);
        if (!groups.has(key)) {
          groups.set(key, []);
        }
        groups.get(key).push(project);
      });

      const map = {};
      groups.forEach(group => {
        group.sort((a, b) => {
          const aTime = new Date(a.createdAt || 0).getTime();
          const bTime = new Date(b.createdAt || 0).getTime();
          if (aTime !== bTime) return aTime - bTime;
          return Number(a.id || 0) - Number(b.id || 0);
        });

        if (group.length === 1) {
          const only = group[0];
          map[only.id] = only.nomProjet || `Projet ${only.id}`;
          return;
        }

        group.forEach((project, index) => {
          const baseName = (project.nomProjet || 'Projet').trim() || 'Projet';
          map[project.id] = `${baseName} ${index + 1}`;
        });
      });

      return map;
    }

    }

    function polarToCartesian(cx, cy, radius, angleInDegrees) {
      const angleInRadians = ((angleInDegrees - 90) * Math.PI) / 180.0;
      return {
        x: cx + radius * Math.cos(angleInRadians),
        y: cy + radius * Math.sin(angleInRadians),
      };
    }

    function describeArc(cx, cy, radius, startAngle, endAngle) {
      const start = polarToCartesian(cx, cy, radius, endAngle);
      const end = polarToCartesian(cx, cy, radius, startAngle);
      const largeArcFlag = endAngle - startAngle <= 180 ? '0' : '1';

      return [
        'M', cx, cy,
        'L', start.x, start.y,
        'A', radius, radius, 0, largeArcFlag, 0, end.x, end.y,
        'Z',
      ].join(' ');
    }

    function renderExpensesByProject(expenses) {
      const tableContainer = document.getElementById('expenses-by-project-table');
      const chartContainer = document.getElementById('expenses-by-project-chart');
      if (!tableContainer || !chartContainer) return;

      const totalsByProject = {};
      expenses.forEach(expense => {
        const linkedProject = getProjectById(expense?.projetId);
        let projectKey = '';
        let projectLabel = '';

        if (linkedProject) {
          projectKey = `project_${linkedProject.id}`;
          const houseLabel = String(linkedProject.numeroMaison || '').trim();
          const displayName = getProjectDisplayName(linkedProject) || linkedProject.nomProjet || `Projet ${linkedProject.id}`;
          projectLabel = houseLabel
            ? `${displayName} - Maison N°${houseLabel}`
            : displayName;
        } else {
          const projectBucket = getExpenseProjectBucket(expense);
          projectKey = projectBucket.key;
          projectLabel = projectBucket.label;
        }

        const amount = Number(expense.montantTotal || 0);
        if (!totalsByProject[projectKey]) {
          totalsByProject[projectKey] = {
            project: projectLabel,
            total: 0,
          };
        }
        totalsByProject[projectKey].total += amount;
      });

      const entries = Object.values(totalsByProject)
        .sort((a, b) => b.total - a.total);

      if (!entries.length) {
        tableContainer.innerHTML = '<p style="color: #64748b;">Aucune dépense enregistrée.</p>';
        chartContainer.innerHTML = '<p style="color: #64748b;">Aucune donnée pour le diagramme.</p>';
        return;
      }

      const grandTotal = entries.reduce((sum, item) => sum + item.total, 0);

      tableContainer.innerHTML = `
        <div class="table-wrapper" style="overflow-x: auto;">
          <table class="table">
            <thead>
              <tr>
                <th>Projet</th>
                <th>Total dépenses (FCFA)</th>
                <th>Part</th>
              </tr>
            </thead>
            <tbody>
              ${entries.map(item => `
                <tr>
                  <td>${escapeHtml(item.project)}</td>
                  <td>${item.total.toFixed(2)}</td>
                  <td>${((item.total / grandTotal) * 100).toFixed(1)}%</td>
                </tr>
              `).join('')}
            </tbody>
          </table>
        </div>
      `;

      const colors = ['#2563eb', '#10b981', '#f59e0b', '#ef4444', '#8b5cf6', '#06b6d4', '#84cc16', '#f97316'];
      const cx = 110;
      const cy = 110;
      const radius = 90;
      let currentAngle = 0;

      const slices = entries.map((item, index) => {
        const angle = (item.total / grandTotal) * 360;
        const startAngle = currentAngle;
        const endAngle = currentAngle + angle;
        currentAngle += angle;
        const color = colors[index % colors.length];

        return `<path d="${describeArc(cx, cy, radius, startAngle, endAngle)}" fill="${color}"></path>`;
      }).join('');

      const legend = entries.map((item, index) => {
        const color = colors[index % colors.length];
        return `
          <div style="display: flex; align-items: center; gap: 8px; margin-bottom: 6px; font-size: 12px;">
            <span style="display:inline-block; width: 10px; height: 10px; border-radius: 50%; background:${color};"></span>
            <span>${escapeHtml(item.project)} (${item.total.toFixed(2)} FCFA)</span>
          </div>
        `;
      }).join('');

      chartContainer.innerHTML = `
        <div style="display: flex; gap: 16px; flex-wrap: wrap; align-items: center;">
          <svg width="220" height="220" viewBox="0 0 220 220" role="img" aria-label="Diagramme circulaire des dépenses par projet">
            ${slices}
            <circle cx="${cx}" cy="${cy}" r="40" fill="#ffffff"></circle>
            <text x="${cx}" y="${cy}" text-anchor="middle" dominant-baseline="middle" style="font-size: 11px; fill: #334155;">${grandTotal.toFixed(0)} FCFA</text>
          </svg>
          <div>${legend}</div>
        </div>
      `;
    }

    function getExpenseProjectBucket(expense) {
      const project = getProjectById(expense?.projetId);
      if (project) {
        const key = getProjectGroupKey(project);
        const projectName = String(project.nomProjet || `Projet ${project.id}`).trim() || `Projet ${project.id}`;
        const projectSite = String(project.nomSite || 'Non renseigne').trim() || 'Non renseigne';
        return {
          key,
          label: `${projectName} - ${projectSite}`,
        };
      }

      const category = normalizeText(expense?.categorie || '');
      const details = normalizeText(expense?.description || expense?.itemName || '');
      const isTransportCost = category === 'transport'
        || details.includes('transport')
        || details.includes('carburant')
        || details.includes('essence');

      const rawName = String(expense?.projetNom || '').trim();
      let fallbackName = rawName.replace(/\s+\d+$/, '').trim() || rawName;
      if (!fallbackName || (normalizeText(fallbackName) === 'sans projet' && isTransportCost)) {
        fallbackName = isTransportCost ? 'Transport' : 'Sans projet';
      }

      return {
        key: `fallback__${normalizeText(fallbackName)}`,
        label: fallbackName,
      };
    }

    function getExpenseCategoryLabel(categoryKey) {

      return 'Materiaux';
    }

    function renderExpensesDetailedByProject(expenses) {
      const container = document.getElementById('expenses-detailed-by-project-table');
      if (!container) return;

      const grouped = {};

      expenses.forEach(expense => {
        const bucket = getExpenseProjectBucket(expense);
        const category = getExpenseCategory(expense.categorie);
        const detailLabel = String(expense.description || expense.itemName || 'Sans precision').trim() || 'Sans precision';
        const amount = Number(expense.montantTotal || 0);

        if (!grouped[bucket.key]) {
          grouped[bucket.key] = {
            label: bucket.label,
            details: {},
            totals: {
              materiaux: 0,
              mainOeuvre: 0,
              salaires: 0,
              total: 0,
            },
          };
        }

        const detailKey = `${category}__${normalizeText(detailLabel)}`;
        if (!grouped[bucket.key].details[detailKey]) {
          grouped[bucket.key].details[detailKey] = {
            category,
            detailLabel,
            amount: 0,
          };
        }

        grouped[bucket.key].details[detailKey].amount += amount;
        grouped[bucket.key].totals[category] += amount;
        grouped[bucket.key].totals.total += amount;
      });

      const projects = Object.values(grouped).sort((a, b) => b.totals.total - a.totals.total);

      if (!projects.length) {
        container.innerHTML = '<p style="color: #64748b;">Aucune dépense enregistrée.</p>';
        return;
      }

      const categoryOrder = { materiaux: 0, mainOeuvre: 1, salaires: 2 };

      let html = `
        <table class="table">
          <thead>
            <tr>
              <th>Projet</th>
              <th>Catégorie</th>
              <th>Détail</th>
              <th>Montant (FCFA)</th>
            </tr>
          </thead>
          <tbody>
      `;

      projects.forEach(project => {
        html += `
          <tr style="background: #f8fafc;">
            <td><strong>${escapeHtml(project.label)}</strong></td>
            <td><strong>Total projet</strong></td>
            <td>
              Matériaux: ${project.totals.materiaux.toFixed(2)} FCFA<br>

            <td><strong>${project.totals.total.toFixed(2)} FCFA</strong></td>
          </tr>
        `;

        const details = Object.values(project.details).sort((a, b) => {
          const categoryCompare = (categoryOrder[a.category] || 99) - (categoryOrder[b.category] || 99);
          if (categoryCompare !== 0) return categoryCompare;

        details.forEach(detail => {
          html += `
            <tr>

    function buildFinancialExpenses(baseExpenses, purchaseOrders) {
      const normalizedExpenses = Array.isArray(baseExpenses) ? [...baseExpenses] : [];
      const orders = Array.isArray(purchaseOrders) ? purchaseOrders : [];

      const existingPurchaseOrderIds = new Set(
        normalizedExpenses
          .map(expense => Number(expense.purchaseOrderId || 0))
          .filter(id => Number.isFinite(id) && id > 0)
      );

      orders.forEach(order => {
        const orderId = Number(order.id || 0);
        const total = Number(order.montantTotal || 0);

        if (!orderId || existingPurchaseOrderIds.has(orderId) || total <= 0) {
          return;
        }

        if (String(order.statut || '').toUpperCase() === 'REJETEE') {
          return;
        }

        normalizedExpenses.push({
          id: `po-${orderId}`,
          purchaseOrderId: orderId,
          projetId: order.projetId || null,
          projetNom: order.nomProjet || 'Sans projet',
          categorie: 'materiaux',
          description: String(order.itemName || '').trim() || `Materiel bon #${orderId}`,
          quantite: 1,
          prixUnitaire: total,
          montantTotal: total,
          statut: order.statut || 'EN_ATTENTE',
        });
      });

      return normalizedExpenses;
    }

    async function fetchJson(url, options = {}) {
      const res = await fetch(url, { headers, ...options });
      if (res.status === 401) {
        localStorage.removeItem('erp_token');
        window.location.href = 'index.html';
      }
      if (!res.ok) {
        const error = await res.json().catch(() => ({}));
        throw new Error(error.error || `Erreur ${res.status}`);
      }
      return res.json();
    }

    async function loadInventory() {
      try {
        const [materials, projects] = await Promise.all([
          fetchJson('/api/material-requests'),
          fetchJson('/api/projects').catch(() => projectsCache),
        ]);
        if (Array.isArray(projects) && projects.length) {
          projectsCache = projects;
        }

        const inventoryMaterials = materials.filter(m => m.statut !== 'REJETEE');
        const list = document.getElementById('inventory-list');
        const summaryList = document.getElementById('inventory-summary-list');

        const folders = buildProjectFolders(projectsCache);
        const inventoryProjectFilter = document.getElementById('inventory-project-filter');
        if (inventoryProjectFilter) {
          const previousFolderKey = inventoryProjectFilter.value || selectedInventoryProjectFolderKey;
          inventoryProjectFilter.innerHTML = '<option value="">Sélectionner un dossier projet</option>' +
            folders.map(folder => `<option value="${folder.key}">${escapeHtml(folder.nomProjet)} - ${escapeHtml(folder.nomSite)}</option>`).join('');

          if (previousFolderKey && folders.some(folder => folder.key === previousFolderKey)) {
            inventoryProjectFilter.value = previousFolderKey;
          } else if (folders.length) {
            inventoryProjectFilter.value = folders[0].key;
          }

          selectedInventoryProjectFolderKey = inventoryProjectFilter.value;
        }

        const inventoryViewInput = document.getElementById('inventory-view-search');
        if (inventoryViewInput && !inventoryViewInput.value) {
          inventoryViewInput.value = inventoryViewSearchTerm;
        }
        const resolvedViewMode = resolveInventoryViewMode(inventoryViewInput ? inventoryViewInput.value : inventoryViewSearchTerm);
        applyInventoryViewMode(resolvedViewMode);

        const selectedFolder = folders.find(folder => folder.key === selectedInventoryProjectFolderKey) || null;
        const selectedProjectIds = new Set((selectedFolder?.houses || []).map(house => Number(house.id)));
        const inventoryProjectTitle = document.getElementById('inventory-project-view-title');
        if (inventoryProjectTitle) {
          inventoryProjectTitle.textContent = selectedFolder
            ? `Tableau - ${selectedFolder.nomProjet || 'Projet'} (${selectedFolder.nomSite || 'Site non renseigne'})`
            : "Tableau - Produits d'un seul projet";
        }

        const projectTableMap = {};
        inventoryMaterials
          .filter(item => selectedProjectIds.has(Number(item.projetId)))
          .forEach(item => {
            const project = getProjectById(item.projetId);
            const houseNumber = String(item.numeroMaison || project?.numeroMaison || '-').trim() || '-';
            const product = item.itemName || 'Materiel';
            const key = `${houseNumber}__${product}`;

            if (!projectTableMap[key]) {
              projectTableMap[key] = {
                houseNumber,
                product,
                totalDemande: 0,
                totalRestant: 0,
                hasStockArrival: false,
              };
            }

            projectTableMap[key].totalDemande += Number(item.quantiteDemandee || 0);
            projectTableMap[key].totalRestant += Number(item.quantiteRestante || 0);
            if (String(item.statut || '').toUpperCase() === 'EN_STOCK') {
              projectTableMap[key].hasStockArrival = true;
            }
          });

        const projectRows = Object.values(projectTableMap)
          .map(row => ({
            ...row,
            totalUtilise: Math.max(row.totalDemande - row.totalRestant, 0),
            statut: row.hasStockArrival
              ? 'EN_STOCK'
              : (row.totalRestant <= 0 && row.totalDemande > 0 ? 'EPUISE' : 'EN_ATTENTE'),
          }))
          .sort((a, b) => {
            const houseCompare = String(a.houseNumber).localeCompare(String(b.houseNumber), 'fr', { numeric: true });
            if (houseCompare !== 0) return houseCompare;
            return String(a.product).localeCompare(String(b.product), 'fr');
          });

        if (list) {
          if (!selectedFolder) {
            list.innerHTML = '<tr><td colspan="6" style="text-align: center; padding: 20px;">Sélectionne un dossier projet pour afficher les produits</td></tr>';
          } else {
            list.innerHTML = projectRows.length
              ? projectRows.map(row => `
                  <tr>
                    <td>Maison N°${escapeHtml(row.houseNumber)}</td>
                    <td>${escapeHtml(row.product)}</td>
                    <td>${row.totalDemande.toFixed(2)}</td>
                    <td>${row.totalRestant.toFixed(2)}</td>
                    <td>${row.totalUtilise.toFixed(2)}</td>
                    <td><span class="badge ${getInventoryBadgeClass(row.statut)}">${formatInventoryStatusLabel(row.statut)}</span></td>
                  </tr>
                `).join('')
              : '<tr><td colspan="6" style="text-align: center; padding: 20px;">Aucune donnée pour ce dossier projet</td></tr>';
          }
        }

        const summaryMap = {};
        inventoryMaterials.forEach(m => {
          const produit = m.itemName || 'Materiel';
          const key = normalizeText(produit);

          if (!summaryMap[key]) {
            summaryMap[key] = {
              produit,
              totalDemande: 0,
              totalRestant: 0,
            };
          }

          const demande = Number(m.quantiteDemandee || 0);
          const restant = Number(m.quantiteRestante || 0);
          summaryMap[key].totalDemande += demande;
          summaryMap[key].totalRestant += restant;
        });

        const summaryRows = Object.values(summaryMap).map(row => {
          const totalUtilise = Math.max(row.totalDemande - row.totalRestant, 0);
          let statut = 'EN_ATTENTE';
          if (row.totalRestant <= 0 && row.totalDemande > 0) {
            statut = 'EPUISE';
          } else if (totalUtilise > 0) {
            statut = 'EN_STOCK';
          }

          return {
            ...row,
            totalUtilise,
            statut,
          };
        }).sort((a, b) => a.produit.localeCompare(b.produit, 'fr'));

        if (summaryList) {
          summaryList.innerHTML = summaryRows.length
            ? summaryRows.map(row => `
                <tr>
                  <td>${row.produit}</td>
                  <td>${row.totalDemande.toFixed(2)}</td>
                  <td>${row.totalRestant.toFixed(2)}</td>
                  <td>${row.totalUtilise.toFixed(2)}</td>
                </tr>
              `).join('')
            : '<tr><td colspan="4" style="text-align: center; padding: 20px;">Aucune donnée</td></tr>';
        }
      } catch (err) {
        showMessage('Erreur', 'error');
      }
    }

    window.validateRequestToPo = async id => {
      const request = materialRequestsCache.find(r => Number(r.id) === Number(id));
      if (!request) {
        showMessage('Demande introuvable', 'error');
        return;
      }

      const fournisseur = prompt('Fournisseur pour ce bon :', 'Fournisseur à définir');
      if (fournisseur === null) return;
      const supplierValue = fournisseur.trim();
      if (!supplierValue) {
        showMessage('Le fournisseur est obligatoire', 'error');
        return;
      }

      const prixPrompt = prompt('Prix unitaire (FCFA) :', '0');
      if (prixPrompt === null) return;
      const prixUnitaire = Number(prixPrompt);
      if (Number.isNaN(prixUnitaire) || prixUnitaire < 0) {
        showMessage('Prix unitaire invalide', 'error');
        return;
      }

      const quantite = Number(request.quantiteRestante || request.quantiteDemandee || 1);

      try {
        await fetchJson('/api/purchase-orders', {
          method: 'POST',
          body: JSON.stringify({
            creePar: document.getElementById('po-created-by').value || 'admin',
            fournisseur: supplierValue,
            dateCommande: document.getElementById('po-date').value || formatDateForInput(),
            items: [
              {
                materialRequestId: request.id,
                article: request.itemName || 'Article',
                quantite,
                prixUnitaire,
                totalLigne: quantite * prixUnitaire,
              },
            ],
          }),
        });

        await fetchJson(`/api/material-requests/${request.id}/statut`, {
          method: 'PATCH',
          body: JSON.stringify({ statut: 'APPROUVEE' }),
        });

        showMessage('Demande validée et bon créé');

        loadInventory();

        showMessage('Erreur: ' + err.message, 'error');

    async function loadDatabaseDocuments() {
      try {
        const documents = await fetchJson('/api/database-documents');
        const title = document.getElementById('database-section-title');
        const list = document.getElementById('database-documents-list');
        const sectionInput = document.getElementById('database-section-search');
        const labels = {
          achats: 'Achats',
          construction: 'Construction',
          comptabilite: 'Comptabilité',
        };
        const sectionLabel = labels[databaseSection] || databaseSection;

        if (title) {
          title.textContent = sectionLabel;
        }
        if (sectionInput) {
          sectionInput.placeholder = `Rechercher dans ${sectionLabel.toLowerCase()}`;
          sectionInput.value = databaseSectionSearchTerms[databaseSection] || '';
        }
        if (!list) return;

        const sectionDocuments = documents.filter(doc => doc.sectionCode === databaseSection);
        const globalSearch = normalizeText(databaseGlobalSearchTerm);
        const sectionSearch = normalizeText(databaseSectionSearchTerms[databaseSection]);
        const filteredDocuments = sectionDocuments.filter(doc => {
          const haystack = normalizeText([
            doc.title,
            doc.fileName,
            doc.sectionLabel,
            sectionLabel,
          ].join(' '));

          const matchesGlobal = !globalSearch || haystack.includes(globalSearch);
          const matchesSection = !sectionSearch || haystack.includes(sectionSearch);
          return matchesGlobal && matchesSection;
        });

        if (!sectionDocuments.length) {
          list.innerHTML = '<div class="empty-state"><p>Aucun PDF archivé dans cette section</p></div>';
          return;
        }

        if (!filteredDocuments.length) {
          list.innerHTML = '<div class="empty-state"><p>Aucun document ne correspond à la recherche</p></div>';
          return;
        }

        list.innerHTML = filteredDocuments.map(doc => `
          <li class="list-item">
            <div class="list-item-content">
              <div class="list-item-title">${escapeHtml(doc.title)}</div>
              <div class="list-item-meta">${escapeHtml(doc.fileName)} • ${formatDateWithSameDayTime(doc.updatedAt)}</div>
            </div>
            <div class="action-buttons">
              <button class="button button-small button-primary" onclick="window.open('${escapeHtml(doc.fileUrl)}', '_blank')">Ouvrir PDF</button>
              <button class="button button-small button-danger" onclick="deleteDatabaseDocument(${doc.id})">Supprimer</button>
            </div>
          </li>
        `).join('');
      } catch (err) {
        showMessage('Erreur lors du chargement de la base de donnée', 'error');
      }
    }

    function fileToBase64(file) {
      return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = () => {
          const result = String(reader.result || '');
          const base64 = result.includes(',') ? result.split(',')[1] : result;
          resolve(base64);
        };
        reader.onerror = reject;
        reader.readAsDataURL(file);
      });
    }

    function renderReportsExpensesPie(expenses) {
      const container = document.getElementById('reports-expenses-pie');
      if (!container) return;

      const totals = {};
      let salariesTotal = 0;

      expenses.forEach(expense => {
        const amount = Number(expense.montantTotal || 0);
        const category = normalizeText(expense.categorie || '');
        if (category === 'salaires' || category === 'salaire') {
          salariesTotal += amount;
          return;
        }
        const key = getExpenseProjectBucket(expense).label;
        totals[key] = (totals[key] || 0) + amount;
      });

      const entries = Object.entries(totals).map(([label, total]) => ({ label, total }));
      if (salariesTotal > 0) {
        entries.push({ label: 'Salaires', total: salariesTotal });
      }

      if (!entries.length) {
        container.innerHTML = '<p style="color: #64748b;">Aucune dépense pour le diagramme.</p>';
        return;
      }

      const total = entries.reduce((sum, entry) => sum + entry.total, 0);
      const colors = ['#1d4ed8', '#0f766e', '#f59e0b', '#dc2626', '#7c3aed', '#0891b2', '#65a30d'];
      let currentAngle = 0;
      const slices = entries.map((entry, index) => {
        const angle = (entry.total / total) * 360;
        const path = describeArc(110, 110, 86, currentAngle, currentAngle + angle);
        currentAngle += angle;
        return `<path d="${path}" fill="${colors[index % colors.length]}"></path>`;
      }).join('');

      const legend = entries.map((entry, index) => `
        <div style="display:flex; align-items:center; gap:8px; margin-bottom:8px; font-size:12px;">
          <span style="width:10px; height:10px; border-radius:999px; background:${colors[index % colors.length]}; display:inline-block;"></span>
          <span>${escapeHtml(entry.label)} (${entry.total.toFixed(2)} FCFA)</span>
        </div>
      `).join('');

      container.innerHTML = `
        <div style="display:flex; gap:18px; align-items:center; flex-wrap:wrap;">
          <svg width="220" height="220" viewBox="0 0 220 220" role="img" aria-label="Diagramme circulaire des dépenses par projet et salaires">
            ${slices}
            <circle cx="110" cy="110" r="38" fill="#ffffff"></circle>
            <text x="110" y="110" text-anchor="middle" dominant-baseline="middle" style="font-size:11px; fill:#334155;">${total.toFixed(0)} FCFA</text>
          </svg>
          <div>${legend}</div>
        </div>
      `;
    }

    function renderRevenueLineChart(revenues) {
      const container = document.getElementById('reports-revenue-line');
      const title = document.getElementById('reports-revenue-chart-title');
      const helper = document.getElementById('reports-revenue-chart-helper');
      if (!container) return;

      const width = 420;
      const height = 220;
      const padding = 34;
      const normalizedSearch = normalizeText(reportsSearchTerm);
      const showTurnoverOnly = normalizedSearch.includes('chiffre') || normalizedSearch.includes('affaire') || normalizedSearch.includes('ca');

      if (title) {
        title.textContent = showTurnoverOnly ? 'Courbe du chiffre d\'affaires' : 'Courbes des recettes';
      }
      if (helper) {
        helper.textContent = showTurnoverOnly
          ? 'Mode recherché: chiffre d\'affaires.'
          : 'Affichage par défaut: recette totale cumulée et recette marginale par projet/contrat enregistré.';
      }

      if (!revenues.length) {
        const yAxis = Array.from({ length: 5 }, (_, index) => {
          const value = index * 25;
          const y = height - padding - (index / 4) * (height - padding * 2);
          return `
                  <div class="list-item-meta">${e.categorie} • ${e.quantite} × ${e.prixUnitaire} FCFA • ${e.projetNom || 'Aucun projet'}</div>
            <text x="${padding - 8}" y="${y + 4}" text-anchor="end" style="font-size:11px; fill:#64748b;">${value}</text>
          `;
        }).join('');

                <div style="font-weight: 600;">${(e.montantTotal || 0).toFixed(2)} FCFA</div>
          <svg width="100%" height="260" viewBox="0 0 ${width} ${height + 30}" role="img" aria-label="Courbe des recettes vide">
            ${yAxis}
            <line x1="${padding}" y1="${height - padding}" x2="${width - padding}" y2="${height - padding}" stroke="#94a3b8"></line>
            <line x1="${padding}" y1="${padding}" x2="${padding}" y2="${height - padding}" stroke="#94a3b8"></line>
            <text x="${width / 2}" y="${height / 2}" text-anchor="middle" style="font-size:13px; fill:#64748b;">Aucun revenu enregistré</text>
            <text x="${width / 2}" y="${height / 2 + 20}" text-anchor="middle" style="font-size:11px; fill:#94a3b8;">Enregistre un revenu pour afficher la courbe</text>
          </svg>
        `;
        return;
        renderExpensesDetailedByProject(expenses);
      }

      const grouped = {};
      revenues.forEach(revenue => {
        const key = formatDateForInput(revenue.dateRevenue || new Date());
        grouped[key] = (grouped[key] || 0) + Number(revenue.amount || 0);
      });

      const points = Object.entries(grouped)
        .sort((a, b) => a[0].localeCompare(b[0]))
        .map(([date, amount]) => ({ date, amount }));

      const cumulativePoints = [];
      let runningTotal = 0;
      points.forEach(point => {
        runningTotal += point.amount;
        cumulativePoints.push({ ...point, amount: runningTotal });
      });

      const maxAmount = Math.max(
        ...(showTurnoverOnly ? points : [...points, ...cumulativePoints]).map(point => point.amount),
        1
      );
      const minAmount = 0;
      const xStep = points.length > 1 ? (width - padding * 2) / (points.length - 1) : 0;
      const yScale = value => height - padding - ((value - minAmount) / (maxAmount - minAmount || 1)) * (height - padding * 2);
      const coords = points.map((point, index) => ({
        ...point,
        x: padding + xStep * index,
        y: yScale(point.amount),
      }));
      const cumulativeCoords = cumulativePoints.map((point, index) => ({
        ...point,
        x: padding + xStep * index,
        y: yScale(point.amount),
      }));

      const linePath = coords.map((point, index) => `${index === 0 ? 'M' : 'L'} ${point.x} ${point.y}`).join(' ');
      const areaPath = `${linePath} L ${coords[coords.length - 1].x} ${height - padding} L ${coords[0].x} ${height - padding} Z`;
      const cumulativePath = cumulativeCoords.map((point, index) => `${index === 0 ? 'M' : 'L'} ${point.x} ${point.y}`).join(' ');
      const xLabels = coords.map(point => `
        <text x="${point.x}" y="${height - 10}" text-anchor="middle" style="font-size:11px; fill:#64748b;">${escapeHtml(point.date.slice(5))}</text>
      `).join('');
      const yTicks = 4;
      const yAxis = Array.from({ length: yTicks + 1 }, (_, index) => {
        const value = (maxAmount / yTicks) * index;
        const y = yScale(value);
        return `
          <line x1="${padding}" y1="${y}" x2="${width - padding}" y2="${y}" stroke="#e2e8f0" stroke-dasharray="4 4"></line>
          <text x="${padding - 8}" y="${y + 4}" text-anchor="end" style="font-size:11px; fill:#64748b;">${value.toFixed(0)}</text>
        `;
      }).join('');
      const dots = coords.map(point => `
        <circle cx="${point.x}" cy="${point.y}" r="4" fill="#2563eb"></circle>
        <title>${point.date}: ${point.amount.toFixed(2)} FCFA</title>
      `).join('');
      const cumulativeDots = cumulativeCoords.map(point => `
        <circle cx="${point.x}" cy="${point.y}" r="4" fill="#10b981"></circle>
        <title>${point.date}: ${point.amount.toFixed(2)} FCFA</title>
      `).join('');
      const legend = showTurnoverOnly
        ? '<div style="display:flex; gap:14px; margin-top:8px; font-size:12px; color:#475569;"><span><span style="display:inline-block;width:10px;height:10px;border-radius:999px;background:#2563eb;margin-right:6px;"></span>Chiffre d\'affaires</span></div>'
        : '<div style="display:flex; gap:14px; margin-top:8px; font-size:12px; color:#475569; flex-wrap: wrap;"><span><span style="display:inline-block;width:10px;height:10px;border-radius:999px;background:#10b981;margin-right:6px;"></span>Recette totale</span><span><span style="display:inline-block;width:10px;height:10px;border-radius:999px;background:#2563eb;margin-right:6px;"></span>Recette marginale</span></div>';

      container.innerHTML = `
        <svg width="100%" height="260" viewBox="0 0 ${width} ${height + 30}" role="img" aria-label="${showTurnoverOnly ? 'Courbe du chiffre d\'affaires' : 'Courbes des recettes'}">
          ${yAxis}
          <line x1="${padding}" y1="${height - padding}" x2="${width - padding}" y2="${height - padding}" stroke="#94a3b8"></line>
          <line x1="${padding}" y1="${padding}" x2="${padding}" y2="${height - padding}" stroke="#94a3b8"></line>
          <path d="${areaPath}" fill="rgba(37,99,235,0.12)"></path>
          ${showTurnoverOnly ? '' : `<path d="${cumulativePath}" fill="none" stroke="#10b981" stroke-width="3" stroke-linecap="round" stroke-linejoin="round"></path>`}
          <path d="${linePath}" fill="none" stroke="#2563eb" stroke-width="3" stroke-linecap="round" stroke-linejoin="round"></path>
          ${showTurnoverOnly ? '' : cumulativeDots}
          ${dots}
          ${xLabels}
        </svg>
        ${legend}
      `;
    }

    async function loadReports() {
      try {
        const [projects, expenses, purchaseOrders, revenues] = await Promise.all([
          fetchJson('/api/projects'),
          fetchJson('/api/expenses'),
          fetchJson('/api/purchase-orders').catch(() => []),
          fetchJson('/api/revenues'),
        ]);
        const financialExpenses = buildFinancialExpenses(expenses, purchaseOrders);

        const projectData = {};
        projectsCache = projects;
        const folders = buildProjectFolders(projects);
        const folderByLegacyName = new Map();

        folders.forEach(folder => {
          const folderLabel = `${folder.nomProjet || 'Projet'} - ${folder.nomSite || 'Non renseigne'}`;
          folderByLegacyName.set(normalizeText(folderLabel), folder);
          folderByLegacyName.set(normalizeText(folder.nomProjet || ''), folder);
        });

        financialExpenses.forEach(expense => {
          let projectBucket = null;
          const linkedProject = getProjectById(expense?.projetId);

          if (linkedProject) {
            projectBucket = getExpenseProjectBucket(expense);
          } else {
            const rawLegacyName = String(expense?.projetNom || '').trim();
            if (rawLegacyName) {
              const normalizedLegacyName = normalizeText(rawLegacyName.replace(/\s+\d+$/, '').trim());
              let matchedFolder = folderByLegacyName.get(normalizedLegacyName);
              if (!matchedFolder && normalizedLegacyName) {
                matchedFolder = folders.find(folder => {
                  const normalizedProjectName = normalizeText(folder.nomProjet || '');
                  const normalizedSite = normalizeText(folder.nomSite || '');
                  const normalizedFolderLabel = normalizeText(`${folder.nomProjet || ''} - ${folder.nomSite || ''}`);

                  if (!normalizedProjectName) return false;

                  return normalizedLegacyName.includes(normalizedFolderLabel)
                    || normalizedFolderLabel.includes(normalizedLegacyName)
                    || normalizedLegacyName.includes(normalizedProjectName)
                    || normalizedProjectName.includes(normalizedLegacyName)
                    || (normalizedSite && normalizedLegacyName.includes(normalizedSite));
                });
              }
              if (matchedFolder) {
                projectBucket = {
                  key: matchedFolder.key,
                  label: `${matchedFolder.nomProjet || 'Projet'} - ${matchedFolder.nomSite || 'Non renseigne'}`,
                };
              }
            }
          }

          if (!projectBucket) {
            return;
          }

          if (!projectData[projectBucket.key]) {
            projectData[projectBucket.key] = {
              nom: projectBucket.label,
              categories: {
                materiaux: {},
                mainOeuvre: {},
                salaires: {}
              },
              totaux: {
                materiaux: 0,
                mainOeuvre: 0,
                salaires: 0,
                total: 0
              }
            };
          }

          const category = getExpenseCategory(expense.categorie);
          const rawDetailLabel = String(expense.description || expense.itemName || 'Sans precision').trim() || 'Sans precision';
          const detailKey = getExpenseDetailKey(rawDetailLabel);
          const detailLabel = getExpenseDetailLabel(rawDetailLabel);
          if (!projectData[projectBucket.key].categories[category][detailKey]) {
            projectData[projectBucket.key].categories[category][detailKey] = {
              label: detailLabel,
              amount: 0
            };
          }

          projectData[projectBucket.key].categories[category][detailKey].amount += expense.montantTotal || 0;
          projectData[projectBucket.key].totaux[category] += expense.montantTotal || 0;
          projectData[projectBucket.key].totaux.total += expense.montantTotal || 0;
        });

        generateExpensesTable(projectData);
        renderReportsExpensesPie(financialExpenses);
        renderRevenueLineChart(revenues);
      } catch (err) {
        console.error('Erreur lors du chargement des rapports:', err);
        showMessage('Erreur lors du chargement des rapports', 'error');
      }
    }

    function getExpenseCategory(categorie) {
      const cat = categorie?.toLowerCase() || '';
      if (['salaires', 'salaire'].includes(cat)) return 'salaires';
      if (['peintre', 'maçon', 'main d\'oeuvre', 'main-d\'oeuvre'].includes(cat)) return 'mainOeuvre';
      return 'materiaux'; // Par défaut
    }

    function getExpenseDetailKey(label) {
      const compact = String(label || '').replace(/\s+/g, ' ').trim();
      if (!compact) return 'sans-precision';
      return normalizeText(compact);
    }

    function getExpenseDetailLabel(label) {
      const compact = String(label || '').replace(/\s+/g, ' ').trim();
      if (!compact) return 'Sans precision';
      const lower = compact.toLocaleLowerCase('fr-FR');
      return lower.charAt(0).toLocaleUpperCase('fr-FR') + lower.slice(1);
    }

    function generateExpensesTable(projectData) {
      const container = document.getElementById('expenses-report-table');

      if (Object.keys(projectData).length === 0) {
        container.innerHTML = '<p>Aucune dépense liée à un projet disponible.</p>';
        return;
      }

      // Collecter toutes les sous-catégories uniques
      const allSubCategories = {
        materiaux: new Map(),
        mainOeuvre: new Map(),
        salaires: new Map()
      };

      Object.values(projectData).forEach(project => {
        Object.entries(project.categories.materiaux).forEach(([key, value]) => allSubCategories.materiaux.set(key, value.label));
        Object.entries(project.categories.mainOeuvre).forEach(([key, value]) => allSubCategories.mainOeuvre.set(key, value.label));
        Object.entries(project.categories.salaires).forEach(([key, value]) => allSubCategories.salaires.set(key, value.label));
      });

      // Créer le tableau HTML
      let html = `
        <table style="border-collapse: collapse; width: 100%; margin-top: 20px;">
          <thead>
            <tr>
              <th style="border: 1px solid #ddd; padding: 12px; background: #f5f5f5; text-align: left;">Poste</th>`;

      // Ajouter les colonnes des projets
      Object.values(projectData).forEach(project => {
        html += `<th style="border: 1px solid #ddd; padding: 12px; background: #f5f5f5; text-align: center;">${project.nom}</th>`;
      });

      html += `</tr></thead><tbody>`;

      // Ligne Matériel (total)
      html += `<tr><td style="border: 1px solid #ddd; padding: 8px; font-weight: bold; background: #fafafa;">Matériel</td>`;
      Object.values(projectData).forEach(project => {
        html += `<td style="border: 1px solid #ddd; padding: 8px; text-align: center; font-weight: bold;">${project.totaux.materiaux.toFixed(2)} FCFA</td>`;
      });
      html += `</tr>`;

      // Sous-catégories matériaux
      Array.from(allSubCategories.materiaux.entries()).forEach(([itemKey, itemLabel]) => {
        html += `<tr><td style="border: 1px solid #ddd; padding: 8px; padding-left: 20px;">${itemLabel}</td>`;
        Object.values(projectData).forEach(project => {
          const amount = project.categories.materiaux[itemKey]?.amount || 0;
          html += `<td style="border: 1px solid #ddd; padding: 8px; text-align: center;">${amount.toFixed(2)} FCFA</td>`;
        });
        html += `</tr>`;
      });

      // Ligne Total Main d'œuvre
      html += `<tr><td style="border: 1px solid #ddd; padding: 8px; font-weight: bold; background: #fafafa;">Total Main d'œuvre</td>`;
      Object.values(projectData).forEach(project => {
        html += `<td style="border: 1px solid #ddd; padding: 8px; text-align: center; font-weight: bold;">${project.totaux.mainOeuvre.toFixed(2)} FCFA</td>`;
      });
      html += `</tr>`;

      // Sous-catégories main d'œuvre
      Array.from(allSubCategories.mainOeuvre.entries()).forEach(([itemKey, itemLabel]) => {
        html += `<tr><td style="border: 1px solid #ddd; padding: 8px; padding-left: 20px;">${itemLabel}</td>`;
        Object.values(projectData).forEach(project => {
          const amount = project.categories.mainOeuvre[itemKey]?.amount || 0;
          html += `<td style="border: 1px solid #ddd; padding: 8px; text-align: center;">${amount.toFixed(2)} FCFA</td>`;
        });
        html += `</tr>`;
      });

      // Ligne Total Salaires
      html += `<tr><td style="border: 1px solid #ddd; padding: 8px; font-weight: bold; background: #fafafa;">Total Salaires</td>`;
      Object.values(projectData).forEach(project => {
        html += `<td style="border: 1px solid #ddd; padding: 8px; text-align: center; font-weight: bold;">${project.totaux.salaires.toFixed(2)} FCFA</td>`;
      });
      html += `</tr>`;

      // Sous-catégories salaires
      Array.from(allSubCategories.salaires.entries()).forEach(([itemKey, itemLabel]) => {
        html += `<tr><td style="border: 1px solid #ddd; padding: 8px; padding-left: 20px;">${itemLabel}</td>`;
        Object.values(projectData).forEach(project => {
          const amount = project.categories.salaires[itemKey]?.amount || 0;
          html += `<td style="border: 1px solid #ddd; padding: 8px; text-align: center;">${amount.toFixed(2)} FCFA</td>`;
        });
        html += `</tr>`;
      });

      // Ligne Total de chaque projet
      html += `<tr><td style="border: 1px solid #ddd; padding: 8px; font-weight: bold; background: #e8f4f8;">Total de chaque projet</td>`;
      Object.values(projectData).forEach(project => {
        html += `<td style="border: 1px solid #ddd; padding: 8px; text-align: center; font-weight: bold; background: #e8f4f8;">${project.totaux.total.toFixed(2)} FCFA</td>`;
      });
      html += `</tr>`;

      // Ligne Total Global
      const totalGlobal = Object.values(projectData).reduce((sum, project) => sum + project.totaux.total, 0);
      html += `<tr><td style="border: 1px solid #ddd; padding: 8px; font-weight: bold; background: #d4edda;">Total Global</td>`;
      html += `<td colspan="${Object.keys(projectData).length}" style="border: 1px solid #ddd; padding: 8px; text-align: center; font-weight: bold; background: #d4edda;">${totalGlobal.toFixed(2)} FCFA</td>`;
      html += `</tr>`;

      html += `</tbody></table>`;

      container.innerHTML = html;
    }
